95 def extract_patch_coordinates(dims, rf, stride):
96  """
97  Function that partition the FOV in patches
98 
99  and return the indexed in 2D and 1D (flatten, order='F') formats
100 
101  Parameters:
102  ----------
103  dims: tuple of int
104  dimensions of the original matrix that will be divided in patches
105 
106  rf: tuple of int
107  radius of receptive field, corresponds to half the size of the square patch
108 
109  stride: tuple of int
110  degree of overlap of the patches
111  """
112  coords_flat = []
113  shapes = []
114  iters = [list(range(rf[i], dims[i] - rf[i], 2 * rf[i] - stride[i])) + [dims[i] - rf[i]]
115  for i in range(len(dims))]
116  coords = np.empty(list(map(len, iters)) + [len(dims)], dtype=np.object)
117  for count_0, xx in enumerate(iters[0]):
118  coords_x = np.arange(xx - rf[0], xx + rf[0] + 1)
119  coords_x = coords_x[(coords_x >= 0) & (coords_x < dims[0])]
120  for count_1, yy in enumerate(iters[1]):
121  coords_y = np.arange(yy - rf[1], yy + rf[1] + 1)
122  coords_y = coords_y[(coords_y >= 0) & (coords_y < dims[1])]
123  if len(dims) == 2:
124  idxs = np.meshgrid(coords_x, coords_y)
125  coords[count_0, count_1] = idxs
126  shapes.append(idxs[0].shape[::-1])
127 
128  coords_ = np.ravel_multi_index(idxs, dims, order='F')
129  coords_flat.append(coords_.flatten())
130  else: # 3D data
131  for count_2, zz in enumerate(iters[2]):
132  coords_z = np.arange(zz - rf[2], zz + rf[2] + 1)
133  coords_z = coords_z[(coords_z >= 0) & (coords_z < dims[2])]
134  idxs = np.meshgrid(coords_x, coords_y, coords_z)
135  shps = idxs[0].shape
136  shapes.append([shps[1],shps[0],shps[2]])
137  coords[count_0, count_1, count_2] = idxs
138  coords_ = np.ravel_multi_index(idxs, dims, order='F')
139  coords_flat.append(coords_.flatten())
140 
141  for i, c in enumerate(coords_flat):
142  assert(len(c) == np.prod(shapes[i]))
143 
144  return map(np.sort, coords_flat), shapes
