33 def com(A, d1, d2):
34  """Calculation of the center of mass for spatial components
35 
36  Inputs:
37  ------
38  A: np.ndarray
39  matrix of spatial components (d x K)
40 
41  d1: int
42  number of pixels in x-direction
43 
44  d2: int
45  number of pixels in y-direction
46 
47  Output:
48  -------
49  cm: np.ndarray
50  center of mass for spatial components (K x 2)
51  """
52  nr = np.shape(A)[-1]
53  Coor = dict()
54  Coor['x'] = np.kron(np.ones((d2, 1)), np.expand_dims(list(range(d1)), axis=1))
55  Coor['y'] = np.kron(np.expand_dims(list(range(d2)), axis=1), np.ones((d1, 1)))
56  cm = np.zeros((nr, 2)) # vector for center of mass
57  cm[:, 0] = old_div(np.dot(Coor['x'].T, A), A.sum(axis=0))
58  cm[:, 1] = old_div(np.dot(Coor['y'].T, A), A.sum(axis=0))
59 
60  return cm

210 def distance_masks(M_s,cm_s,max_dist):
211  """
212  Compute distance matrix based on an intersection over union metric. Matrix are compared in order,
213  with matrix i compared with matrix i+1
214 
215  Parameters:
216  ----------
217  M_s: tuples of 1-D arrays
218  The thresholded A matrices (masks) to compare, output of threshold_components
219 
220  cm_s: list of list of 2-ples
221  the centroids of the components in each M_s
222 
223  max_dist: float
224  maximum distance among centroids allowed between components. This corresponds to a distance
225  at which two components are surely disjoined
226 
227  Returns:
228  --------
229  D_s: list of matrix distances
230 
231  Raise:
232  ------
233  Exception('Nan value produced. Error in inputs')
234 
235 
236  """
237  D_s=[]
238 
239  for gt_comp,test_comp,cmgt_comp,cmtest_comp in zip(M_s[:-1],M_s[1:],cm_s[:-1],cm_s[1:]):
240 
241  #todo : better with a function that calls itself
242  print('New Pair **')
243  #not to interfer with M_s
244  gt_comp= gt_comp.copy()[:,:]
245  test_comp= test_comp.copy()[:,:]
246 
247  #the number of components for each
248  nb_gt=np.shape(gt_comp)[-1]
249  nb_test=np.shape(test_comp)[-1]
250  D = np.ones((nb_gt,nb_test))
251 
252  cmgt_comp=np.array(cmgt_comp)
253  cmtest_comp=np.array(cmtest_comp)
254  for i in range(nb_gt):
255  #for each components of gt
256  k=gt_comp[:,np.repeat(i,nb_test)]+test_comp
257  # k is correlation matrix of this neuron to every other of the test
258  for j in range(nb_test): #for each components on the tests
259  dist = np.linalg.norm(cmgt_comp[i]-cmtest_comp[j])
260  # we compute the distance of this one to the other ones
261  if dist<max_dist:
262  # union matrix of the i-th neuron to the jth one
263  union = k[:,j].sum()
264  # we could have used OR for union and AND for intersection while converting
265  # the matrice into real boolean before
266 
267  # product of the two elements' matrices
268  # we multiply the boolean values from the jth omponent to the ith
269  intersection= np.array(gt_comp[:,i].T.dot(test_comp[:,j]).todense()).squeeze()
270 
271  # if we don't have even a union this is pointless
272  if union > 0:
273 
274  #intersection is removed from union since union contains twice the overlaping area
275  #having the values in this format 0-1 is helpfull for the hungarian algorithm that follows
276  D[i,j] = 1-1.*intersection/(union-intersection)
277  else:
278  D[i,j] = 1.
279 
280  if np.isnan(D[i,j]):
281  raise Exception('Nan value produced. Error in inputs')
282  else:
283  D[i,j] = 1
284 
285  D_s.append(D)
286  return D_s

291 def find_matches(D_s, print_assignment=False):
292  #todo todocument
293 
294  matches=[]
295  costs=[]
296  t_start=time.time()
297  for ii,D in enumerate(D_s):
298  #we make a copy not to set changes in the original
299  DD=D.copy()
300  if np.sum(np.where(np.isnan(DD)))>0:
301  raise Exception('Distance Matrix contains NaN, not allowed!')
302 
303  #we do the hungarian
304  indexes = linear_sum_assignment(DD)
305  indexes2=[(ind1,ind2) for ind1,ind2 in zip(indexes[0],indexes[1])]
306  matches.append(indexes)
307  DD=D.copy()
308  total = []
309  #we want to extract those informations from the hungarian algo
310  for row, column in indexes2:
311  value = DD[row,column]
312  if print_assignment:
313  print(('(%d, %d) -> %f' % (row, column, value)))
314  total.append(value)
315  print(('FOV: %d, shape: %d,%d total cost: %f' % (ii, DD.shape[0],DD.shape[1], np.sum(total))))
316  print((time.time()-t_start))
317  costs.append(total)
318  # send back the results in the format we want
319  return matches,costs

325  """
326  Link neurons from different FOVs given matches and costs obtained from the hungarian algorithm
327 
328  Parameters:
329  ----------

331  output of the find_matches function
332 
333  costs: list of lists of scalars

343  Returns:

346 
347  """

358  rows,cols=match

362  neuron.append(cols[m_neur])

368  num_neurons+=1

424 def nf_read_roi(fileobj):
425  """
426  points = read_roi(fileobj)
427 
428  Read ImageJ's ROI format
429 
430  Addapted from https://gist.github.com/luispedro/3437255
431  Copyright: Luis Pedro Coelho <luis@luispedro.org>, 2012
432  License: MIT
433  This is based on:
434  http://rsbweb.nih.gov/ij/developer/source/ij/io/RoiDecoder.java.html
435  http://rsbweb.nih.gov/ij/developer/source/ij/io/RoiEncoder.java.html
436  """
437  SUB_PIXEL_RESOLUTION = 128
438 
439  pos = [4]
440  def get8():
441  pos[0] += 1
442  s = fileobj.read(1)
443  if not s:
444  raise IOError('readroi: Unexpected EOF')
445  return ord(s)
446 
447  def get16():
448  b0 = get8()
449  b1 = get8()
450  return (b0 << 8) | b1
451 
452  def get32():
453  s0 = get16()
454  s1 = get16()
455  return (s0 << 16) | s1
456 
457  def getfloat():
458  v = np.int32(get32())
459  return v.view(np.float32)
460 
461  magic = fileobj.read(4)
462  if magic != 'Iout':
463  raise IOError('Magic number not found')
464 
465  # It seems that the roi type field occupies 2 Bytes, but only one is used
466  get8()
467  top = get16()
468  left = get16()
469 
470  n_coordinates = get16()
471  subtype = get16()
472  if subtype != 0:
473  raise ValueError('roireader: ROI subtype %s not supported (!= 0)' % subtype)
474  options = get16()
475 
476  if options & SUB_PIXEL_RESOLUTION:
477  getc = getfloat
478  points = np.empty((n_coordinates, 2), dtype=np.float32)
479  else:
480  getc = get16
481  points = np.empty((n_coordinates, 2), dtype=np.int16)
482  points[:,1] = [getc() for i in range(n_coordinates)]
483  points[:,0] = [getc() for i in range(n_coordinates)]
484  points[:,1] += left
485  points[:,0] += top
486  points -= 1
487 
488  return points

553 def extract_binary_masks_blob(A, neuron_radius,dims,num_std_threshold=1, minCircularity= 0.5,
554  minInertiaRatio = 0.2,minConvexity = .8):
555  """
556  Function to extract masks from data. It will also perform a preliminary selectino of good masks based on criteria like shape and size
557 
558  Parameters:
559  ----------
560  A: scipy.sparse matris
561  contains the components as outputed from the CNMF algorithm
562 
563  neuron_radius: float
564  neuronal radius employed in the CNMF settings (gSiz)
565 
566  num_std_threshold: int
567  number of times above iqr/1.349 (std estimator) the median to be considered as threshold for the component
568 
569  minCircularity: float
570  parameter from cv2.SimpleBlobDetector
571 
572  minInertiaRatio: float
573  parameter from cv2.SimpleBlobDetector
574 
575  minConvexity: float
576  parameter from cv2.SimpleBlobDetector
577 
578  Returns:
579  --------
580  masks: np.array
581 
582  pos_examples:
583 
584  neg_examples:
585 
586  """
587  import cv2
588  params = cv2.SimpleBlobDetector_Params()
589  params.minCircularity = minCircularity
590  params.minInertiaRatio = minInertiaRatio
591  params.minConvexity = minConvexity
592 
593  # Change thresholds
594  params.blobColor=255
595 
596  params.minThreshold = 0
597  params.maxThreshold = 255
598  params.thresholdStep= 3
599 
600  params.minArea = np.pi*((neuron_radius*.75)**2)
601 
602  params.filterByColor = True
603  params.filterByArea = True
604  params.filterByCircularity = True
605  params.filterByConvexity = True
606  params.filterByInertia = True
607 
608  detector = cv2.SimpleBlobDetector_create(params)
609 
610  masks_ws=[]
611  pos_examples=[]
612  neg_examples=[]
613 
614 
615  for count,comp in enumerate(A.tocsc()[:].T):
616 
617  print(count)
618  comp_d=np.array(comp.todense())
619  gray_image=np.reshape(comp_d,dims,order='F')
620  gray_image=(gray_image-np.min(gray_image))/(np.max(gray_image)-np.min(gray_image))*255
621  gray_image=gray_image.astype(np.uint8)
622 
623  # segment using watershed
624  markers = np.zeros_like(gray_image)
625  elevation_map = sobel(gray_image)
626  thr_1=np.percentile(gray_image[gray_image>0],50)
627  iqr=np.diff(np.percentile(gray_image[gray_image>0],(25,75)))
628  thr_2=thr_1 + num_std_threshold*iqr/1.35
629  markers[gray_image < thr_1] = 1
630  markers[gray_image > thr_2] = 2
631  edges = watershed(elevation_map, markers)-1
632  # only keep largest object
633  label_objects, nb_labels = ndi.label(edges)
634  sizes = np.bincount(label_objects.ravel())
635 
636  if len(sizes)>1:
637  idx_largest = np.argmax(sizes[1:])
638  edges=(label_objects==(1+idx_largest))
639  edges=ndi.binary_fill_holes(edges)
640  else:
641  print('empty component')
642  edges=np.zeros_like(edges)
643 
644  if 1:
645  masks_ws.append(edges)
646  keypoints = detector.detect((edges*200.).astype(np.uint8))
647  else:
648  masks_ws.append(gray_image)
649  keypoints = detector.detect(gray_image)
650 
651  if len(keypoints)>0:
652  pos_examples.append(count)
653 
654  else:
655  neg_examples.append(count)
656 
657  return np.array(masks_ws),np.array(pos_examples),np.array(neg_examples)

694 def extractROIsFromPCAICA(spcomps, numSTD=4, gaussiansigmax=2 , gaussiansigmay=2,thresh=None):
695  """
696  Given the spatial components output of the IPCA_stICA function extract possible regions of interest
697 
698  The algorithm estimates the significance of a components by thresholding the components after gaussian smoothing
699 
700  Parameters:
701  -----------
702 
703  spcompomps, 3d array containing the spatial components
704 
705  numSTD: number of standard deviation above the mean of the spatial component to be considered signiificant
706  """
707 
708  numcomps, width, height=spcomps.shape
709 
710  allMasks=[]
711  maskgrouped=[]
712  for k in range(0,numcomps):
713  comp=spcomps[k]
714  comp=gaussian_filter(comp,[gaussiansigmay,gaussiansigmax])
715 
716  q75, q25 = np.percentile(comp, [75 ,25])
717  iqr = q75 - q25
718  minCompValuePos=np.median(comp)+numSTD*iqr/1.35
719  minCompValueNeg=np.median(comp)-numSTD*iqr/1.35
720 
721  # got both positive and negative large magnitude pixels
722  if thresh is None:
723  compabspos=comp*(comp>minCompValuePos)-comp*(comp<minCompValueNeg)
724  else:
725  compabspos=comp*(comp>thresh)-comp*(comp<-thresh)
726 
727  labeledpos, n = label(compabspos>0, np.ones((3,3)))
728  maskgrouped.append(labeledpos)
729  for jj in range(1,n+1):
730  tmp_mask=np.asarray(labeledpos==jj)
731  allMasks.append(tmp_mask)
732 
733  return allMasks,maskgrouped

736 def detect_duplicates(file_name,dist_thr = 0.1, FOV = (512,512)):
737  """
738  Removes duplicate ROIs from file file_name
739 
740  Parameters:
741  -----------
742  file_name: .zip file with all rois
743 
744  dist_thr: distance threshold for duplicate detection
745 
746  FOV: dimensions of the FOV
747 
748  Returns:
749  --------
750  ind : list of indeces with duplicate entries
751 
752  ind_keep : list of kept indeces
753  """
754  rois = nf_read_roi_zip(file_name,FOV)
755  cm = [scipy.ndimage.center_of_mass(mm) for mm in rois]
756  sp_rois = scipy.sparse.csc_matrix(np.reshape(rois,(rois.shape[0],np.prod(FOV))).T)
757  D = distance_masks([sp_rois,sp_rois],[cm,cm], 10)[0]
758  np.fill_diagonal(D,1)
759  indeces = np.where(D<dist_thr) # pairs of duplicate indeces
760  ind = list(np.unique(indeces[1][indeces[1]>indeces[0]]))
761  ind_keep = list(set(range(D.shape[0]))-set(ind))
762  duplicates = list(np.unique(np.concatenate((indeces[0],indeces[1]))))
763 
764  return duplicates, ind_keep
