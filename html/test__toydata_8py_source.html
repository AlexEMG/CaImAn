12  gamma = .9 # calcium decay time constant
13  np.random.seed(5)
14  centers = np.asarray([[np.random.randint(4, x - 4) for x in dims] for i in range(N)])
15  trueA = np.zeros(dims + (N,), dtype=np.float32)
16  trueS = np.random.rand(N, T) < firerate / float(framerate)
17  trueS[:, 0] = 0
18  trueC = trueS.astype(np.float32)
19  for i in range(1, T):
20  trueC[:, i] += gamma * trueC[:, i - 1]
21  for i in range(N):
22  trueA[tuple(centers[i]) + (i,)] = 1.
23  tmp = np.zeros(dims)
24  tmp[tuple(d / 2 for d in dims)] = 1.
25  z = np.linalg.norm(gaussian_filter(tmp, sig).ravel())
26  trueA = 10 * gaussian_filter(trueA, sig + (0,)) / z
27  Yr = bkgrd + noise * np.random.randn(*(np.prod(dims), T)) + \
28  trueA.reshape((-1, 4), order='F').dot(trueC)
29  return Yr, trueC, trueS, trueA, centers, dims
