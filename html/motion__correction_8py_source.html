500 def motion_correct_iteration(img,template,frame_num,max_shift_w=25,
501  max_shift_h=25,bilateral_blur=False,diameter=10,sigmaColor=10000,sigmaSpace=0):
502 #todo todocument
503  h_i, w_i = template.shape
504  ms_h = max_shift_h
505  ms_w = max_shift_w
506 
507  if bilateral_blur:
508  img=cv2.bilateralFilter(img,diameter,sigmaColor,sigmaSpace)
509  templ_crop=template[max_shift_h:h_i-max_shift_h,max_shift_w:w_i-max_shift_w].astype(np.float32)
510  res = cv2.matchTemplate(img,templ_crop,cv2.TM_CCORR_NORMED)
511 
512  top_left = cv2.minMaxLoc(res)[3]
513  avg_corr=np.max(res)
514  sh_y,sh_x = top_left
515 
516  if (0 < top_left[1] < 2 * ms_h-1) & (0 < top_left[0] < 2 * ms_w-1):
517  # if max is internal, check for subpixel shift using gaussian
518  # peak registration
519  log_xm1_y = np.log(res[sh_x-1,sh_y])
520  log_xp1_y = np.log(res[sh_x+1,sh_y])
521  log_x_ym1 = np.log(res[sh_x,sh_y-1])
522  log_x_yp1 = np.log(res[sh_x,sh_y+1])
523  four_log_xy = 4*np.log(res[sh_x,sh_y])
524 
525  sh_x_n = -(sh_x - ms_h + old_div((log_xm1_y - log_xp1_y), (2 * log_xm1_y - four_log_xy + 2 * log_xp1_y)))
526  sh_y_n = -(sh_y - ms_w + old_div((log_x_ym1 - log_x_yp1), (2 * log_x_ym1 - four_log_xy + 2 * log_x_yp1)))
527  else:
528  sh_x_n = -(sh_x - ms_h)
529  sh_y_n = -(sh_y - ms_w)
530 
531  M = np.float32([[1,0,sh_y_n],[0,1,sh_x_n]])
532  min_,max_ = np.min(img),np.max(img)
533  new_img = np.clip(cv2.warpAffine(img,M,(w_i,h_i),flags=cv2.INTER_CUBIC),min_,max_)
534 
535  new_templ=template*frame_num/(frame_num + 1) + 1./(frame_num + 1)*new_img
536  shift=[sh_x_n,sh_y_n]
537 
538  return new_img,new_templ,shift,avg_corr
