288 def sparseNMF(Y_ds, nr, max_iter_snmf=500, alpha=10e2, sigma_smooth=(.5, .5, .5), remove_baseline=True, perc_baseline=20, nb=1, truncate = 2 ):
289  """
290  Initilaization using sparse NMF
291 
292  Parameters:
293  -----------
294 
295  max_iter_snm: int
296  number of iterations
297 
298  alpha_snmf:
299  sparsity regularizer
300 
301  sigma_smooth_snmf:
302  smoothing along z,x, and y (.5,.5,.5)
303 
304  perc_baseline_snmf:
305  percentile to remove frmo movie before NMF
306 
307  nb: int
308  Number of background components
309 
310  Returns:
311  -------
312 
313  A: np.array
314  2d array of size (# of pixels) x nr with the spatial components. Each column is
315  ordered columnwise (matlab format, order='F')
316 
317  C: np.array
318  2d array of size nr X T with the temporal components
319 
320  center: np.array
321  2d array of size nr x 2 [ or 3] with the components centroids
322  """
323  m = scipy.ndimage.gaussian_filter(np.transpose(
324  Y_ds, [2, 0, 1]), sigma=sigma_smooth, mode='nearest', truncate=truncate)
325  if remove_baseline:
326  bl = np.percentile(m, perc_baseline, axis=0)
327  m1 = np.maximum(0, m - bl)
328  else:
329  bl = 0
330  m1 = m
331 
332  mdl = NMF(n_components=nr, verbose=False, init='nndsvd', tol=1e-10,
333  max_iter=max_iter_snmf, shuffle=True, alpha=alpha, l1_ratio=1)
334  T, d1, d2 = np.shape(m1)
335  d = d1 * d2
336  yr = np.reshape(m1, [T, d], order='F')
337  C = mdl.fit_transform(yr).T
338  A = mdl.components_.T
339  ind_good = np.where(np.logical_and((np.sum(A, 0) * np.std(C, axis=1))
340  > 0, np.sum(A > np.mean(A), axis=0) < old_div(d, 3)))[0]
341 
342  ind_bad = np.where(np.logical_or((np.sum(A, 0) * np.std(C, axis=1))
343  == 0, np.sum(A > np.mean(A), axis=0) > old_div(d, 3)))[0]
344  A_in = np.zeros_like(A)
345 
346  C_in = np.zeros_like(C)
347  A_in[:, ind_good] = A[:, ind_good]
348  C_in[ind_good, :] = C[ind_good, :]
349  A_in = A_in * (A_in > (.1 * np.max(A_in, axis=0))[np.newaxis, :])
350  A_in[:3, ind_bad] = .0001
351  C_in[ind_bad, :3] = .0001
352 
353  m1 = yr.T - A_in.dot(C_in) + np.maximum(0, bl.flatten())[:, np.newaxis]
354  model = NMF(n_components=nb, init='random', random_state=0, max_iter=max_iter_snmf)
355  b_in = model.fit_transform(np.maximum(m1, 0))
356  f_in = model.components_.squeeze()
357  center = caiman.base.rois.com(A_in, d1, d2)
358 
359  return A_in, C_in, center, b_in, f_in

364 def greedyROI(Y, nr=30, gSig=[5, 5], gSiz=[11, 11], nIter=5, kernel=None, nb=1):
365  """
366  Greedy initialization of spatial and temporal components using spatial Gaussian filtering
367 
368  Parameters:
369  --------
370 
371  Y: np.array
372  3d or 4d array of fluorescence data with time appearing in the last axis.
373 
374  nr: int
375  number of components to be found
376 
377  gSig: scalar or list of integers
378  standard deviation of Gaussian kernel along each axis
379 
380  gSiz: scalar or list of integers
381  size of spatial component
382 
383  nIter: int
384  number of iterations when refining estimates
385 
386  kernel: np.ndarray
387  User specified kernel to be used, if present, instead of Gaussian (default None)
388 
389  nb: int
390  Number of background components
391 
392  Returns:
393  -------
394 
395  A: np.array
396  2d array of size (# of pixels) x nr with the spatial components. Each column is
397  ordered columnwise (matlab format, order='F')
398 
399  C: np.array
400  2d array of size nr X T with the temporal components
401 
402  center: np.array
403  2d array of size nr x 2 [ or 3] with the components centroids
404 
405  @Author: Eftychios A. Pnevmatikakis based on a matlab implementation by Yuanjun Gao
406  Simons Foundation, 2015
407 
408  See Also:
409  -------
410  http://www.cell.com/neuron/pdf/S0896-6273(15)01084-3.pdf
411 
412 
413  """
414  print("Greedy initialization of spatial and temporal components using spatial Gaussian filtering")
415  d = np.shape(Y)
416  med = np.median(Y, axis=-1)
417  Y = Y - med[..., np.newaxis]
418  gHalf = np.array(gSiz) // 2
419  gSiz = 2 * gHalf + 1
420  #we initialize every values to zero
421  A = np.zeros((np.prod(d[0:-1]), nr))
422  C = np.zeros((nr, d[-1]))
423  center = np.zeros((nr, Y.ndim - 1))
424 
425  rho = imblur(Y, sig=gSig, siz=gSiz, nDimBlur=Y.ndim - 1, kernel=kernel)
426  v = np.sum(rho**2, axis=-1)
427 
428  for k in range(nr):
429  #we take the highest value of the blurred total image and we define it as the center of the neuron
430  ind = np.argmax(v)
431  ij = np.unravel_index(ind, d[0:-1])
432  for c, i in enumerate(ij):
433  center[k, c] = i
434 
435  #we define a squared size around it
436  ijSig = [[np.maximum(ij[c] - gHalf[c], 0), np.minimum(ij[c] + gHalf[c] + 1, d[c])]
437  for c in range(len(ij))]
438  #we create an array of it (fl like) and compute the trace like the pixel ij trough time
439  dataTemp = np.array(Y[[slice(*a) for a in ijSig]].copy(), dtype=np.float)
440  traceTemp = np.array(np.squeeze(rho[ij]), dtype=np.float)
441 
442  coef, score = finetune(dataTemp, traceTemp, nIter=nIter)
443  C[k, :] = np.squeeze(score)
444  dataSig = coef[..., np.newaxis] * score.reshape([1] * (Y.ndim - 1) + [-1])
445  xySig = np.meshgrid(*[np.arange(s[0], s[1]) for s in ijSig], indexing='xy')
446  arr = np.array([np.reshape(s, (1, np.size(s)), order='F').squeeze()
447  for s in xySig], dtype=np.int)
448  indeces = np.ravel_multi_index(arr, d[0:-1], order='F')
449 
450  A[indeces, k] = np.reshape(coef, (1, np.size(coef)), order='C').squeeze()
451  Y[[slice(*a) for a in ijSig]] -= dataSig.copy()
452  if k < nr - 1:
453  Mod = [[np.maximum(ij[c] - 2 * gHalf[c], 0),
454  np.minimum(ij[c] + 2 * gHalf[c] + 1, d[c])] for c in range(len(ij))]
455  ModLen = [m[1] - m[0] for m in Mod]
456  Lag = [ijSig[c] - Mod[c][0] for c in range(len(ij))]
457  dataTemp = np.zeros(ModLen)
458  dataTemp[[slice(*a) for a in Lag]] = coef
459  dataTemp = imblur(dataTemp[..., np.newaxis], sig=gSig, siz=gSiz, kernel=kernel)
460  temp = dataTemp * score.reshape([1] * (Y.ndim - 1) + [-1])
461  rho[[slice(*a) for a in Mod]] -= temp.copy()
462  v[[slice(*a) for a in Mod]] = np.sum(
463  rho[[slice(*a) for a in Mod]]**2, axis=-1)
464 
465  res = np.reshape(Y, (np.prod(d[0:-1]), d[-1]), order='F') + med.flatten(order='F')[:, None]
466  model = NMF(n_components=nb, init='random', random_state=0)
467  b_in = model.fit_transform(np.maximum(res, 0))
468  f_in = model.components_.squeeze()
469 
470  return A, C, center, b_in, f_in

475 def finetune(Y, cin, nIter=5):
476  """compute a initialized version of A and C
477 
478  Parameters :
479  -----------
480 
481  Y: D1*d2*T*K patches
482 
483  c: array T*K
484  the inital calcium traces
485 
486  nIter: int
487  True indicates that time is listed in the last axis of Y (matlab format)
488  and moves it in the front
489 
490  Returns :
491  --------
492 
493  a: array (d1,D2) the computed A as l2(Y*C)/Y*C
494 
495  c: array(T) C as the sum of As on x*y axis
496 
497 
498  See Also:
499  ---------
500 
501  """
502  #\bug
503  #\warning
504  debug_ = False
505  if debug_:
506  import os
507  f = open('_LOG_1_' + str(os.getpid()), 'w+')
508  f.write('Y:' + str(np.mean(Y)) + '\n')
509  f.write('cin:' + str(np.mean(cin)) + '\n')
510  f.close()
511 
512  #we compute the multiplication of patches per traces ( non negatively )
513  for _ in range(nIter):
514  a = np.maximum(np.dot(Y, cin), 0)
515  a = old_div(a, np.sqrt(np.sum(a**2))) #compute the l2/a
516  cin = np.sum(Y * a[..., np.newaxis], tuple(np.arange(Y.ndim - 1))) #c as the variation of thoses patches
517 
518  return a, cin

597 def hals(Y, A, C, b, f, bSiz=3, maxIter=5):
598  """ Hierarchical alternating least square method for solving NMF problem
599 
600  Y = A*C + b*f
601 
602  input:
603  ------
604  Y: d1 X d2 [X d3] X T, raw data.
605  It will be reshaped to (d1*d2[*d3]) X T in this
606  function
607 
608  A: (d1*d2[*d3]) X K, initial value of spatial components
609 
610  C: K X T, initial value of temporal components
611 
612  b: (d1*d2[*d3]) X nb, initial value of background spatial component
613 
614  f: nb X T, initial value of background temporal component
615 
616  bSiz: int or tuple of int
617  blur size. A box kernel (bSiz X bSiz [X bSiz]) (if int) or bSiz (if tuple) will
618  be convolved with each neuron's initial spatial component, then all nonzero
619  pixels will be picked as pixels to be updated, and the rest will be
620  forced to be 0.
621 
622  maxIter: maximum iteration of iterating HALS.
623 
624  output:
625  -------
626  the updated A, C, b, f
627 
628  @Author: Johannes Friedrich, Andrea Giovannucci
629 
630  See Also:
631  http://proceedings.mlr.press/v39/kimura14.pdf
632  """
633 
634  # smooth the components
635  dims, T = np.shape(Y)[:-1], np.shape(Y)[-1]
636  K = A.shape[1] # number of neurons
637  nb = b.shape[1] # number of background components
638  if isinstance(bSiz, (int, float)):
639  bSiz = [bSiz] * len(dims)
640  ind_A = nd.filters.uniform_filter(np.reshape(A, dims + (K,), order='F'), size=bSiz + [0])
641  ind_A = np.reshape(ind_A > 1e-10, (np.prod(dims), K), order='F')
642  ind_A = spr.csc_matrix(ind_A) # indicator of nonnero pixels
643 
644  def HALS4activity(Yr, A, C, iters=2):
645  U = A.T.dot(Yr)
646  V = A.T.dot(A)
647  for _ in range(iters):
648  for m in range(len(U)): # neurons and background
649  C[m] = np.clip(C[m] + (U[m] - V[m].dot(C)) / V[m, m], 0, np.inf)
650  return C
651 
652  def HALS4shape(Yr, A, C, iters=2):
653  U = C.dot(Yr.T)
654  V = C.dot(C.T)
655  for _ in range(iters):
656  for m in range(K): # neurons
657  ind_pixels = np.squeeze(ind_A[:, m].toarray())
658  A[ind_pixels, m] = np.clip(A[ind_pixels, m] +
659  ((U[m, ind_pixels] - V[m].dot(A[ind_pixels].T)) /
660  V[m, m]), 0, np.inf)
661  for m in range(nb): # background
662  A[:, K + m] = np.clip(A[:, K + m] + ((U[K + m] - V[K + m].dot(A.T)) /
663  V[K + m, K + m]), 0, np.inf)
664  return A
665 
666  Ab = np.c_[A, b]
667  Cf = np.r_[C, f.reshape(nb, -1)]
668  for _ in range(maxIter):
669  Cf = HALS4activity(np.reshape(Y, (np.prod(dims), T), order='F'), Ab, Cf)
670  Ab = HALS4shape(np.reshape(Y, (np.prod(dims), T), order='F'), Ab, Cf)
671 
672  return Ab[:, :-nb], Cf[:-nb], Ab[:, -nb:], Cf[-nb:].reshape(nb, -1)
