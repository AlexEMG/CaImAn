249 def HALS4shape_bckgrnd(Y_resf, B, F, ind_B, iters=5):
250  K = B.shape[-1]
251  U = Y_resf.T
252  V = F.dot(F.T)
253  for _ in range(iters):
254  for m in range(K): # neurons
255  ind_pixels = ind_B[m]
256 
257  B[ind_pixels, m] = np.clip(B[ind_pixels, m] +

259  V[m, m]), 0, np.inf)

368 def determine_search_location(A, dims, method='ellipse', min_size=3, max_size=8, dist=3,
369  expandCore=iterate_structure(generate_binary_structure(2, 1), 2).astype(int), dview=None):
370  """
371  compute the indices of the distance from the cm to search for the spatial component
372 
373  does it by following an ellipse from the cm or doing a step by step dilatation around the cm
374 
375 
376  Parameters:
377  ----------
378  [parsed]
379  cm[i]:
380  center of mass of each neuron
381 
382  A[:, i]: the A of each components
383 
384  dims:
385  the dimension of each A's ( same usually )
386 
387  dist:
388  computed distance matrix
389 
390  dims: [optional] tuple
391  x, y[, z] movie dimensions
392 
393  method: [optional] string
394  method used to expand the search for pixels 'ellipse' or 'dilate'
395 
396  expandCore: [optional] scipy.ndimage.morphology
397  if method is dilate this represents the kernel used for expansion
398 
399  min_size: [optional] int
400 
401  max_size: [optional] int
402 
403  dist: [optional] int
404 
405  dims: [optional] tuple
406  x, y[, z] movie dimensions
407 
408  Returns:
409  --------
410  dist_indicator: np.ndarray
411  distance from the cm to search for the spatial footprint
412 
413  Raise:
414  -------
415  Exception('You cannot pass empty (all zeros) components!')
416  """
417 
418  from scipy.ndimage.morphology import grey_dilation
419 
420  # we initialize the values
421  if len(dims) == 2:
422  d1, d2 = dims
423  elif len(dims) == 3:
424  d1, d2, d3 = dims
425  d, nr = np.shape(A)
426  A = csc_matrix(A)
427  dist_indicator = False * np.ones((d, nr))
428 
429  if method == 'ellipse':
430  Coor = dict()
431  # we create a matrix of size A.x of each pixel coordinate in A.y and inverse
432  if len(dims) == 2:
433  Coor['x'] = np.kron(np.ones(d2), list(range(d1)))
434  Coor['y'] = np.kron(list(range(d2)), np.ones(d1))
435  elif len(dims) == 3:
436  Coor['x'] = np.kron(np.ones(d3 * d2), list(range(d1)))
437  Coor['y'] = np.kron(np.kron(np.ones(d3), list(range(d2))), np.ones(d1))
438  Coor['z'] = np.kron(list(range(d3)), np.ones(d2 * d1))
439  if not dist == np.inf: # determine search area for each neuron
440  cm = np.zeros((nr, len(dims))) # vector for center of mass
441  Vr = [] # cell(nr,1);
442  dist_indicator = []
443  pars = []
444  # for each dim
445  for i, c in enumerate(['x', 'y', 'z'][:len(dims)]):
446  # mass center in this dim = (coor*A)/sum(A)
447  cm[:, i] = old_div(np.dot(Coor[c], A[:, :nr].todense()), A[:, :nr].sum(axis=0))
448 
449  # parrallelizing process of the construct ellipse function
450  for i in range(nr):
451  pars.append([Coor, cm[i], A[:, i], Vr, dims, dist, max_size, min_size, d])
452  if dview is None:
453  res = list(map(construct_ellipse_parallel, pars))
454  else:
455  res = dview.map_sync(construct_ellipse_parallel, pars)
456  for r in res:
457  dist_indicator.append(r)
458 
459  dist_indicator = (np.asarray(dist_indicator)).squeeze().T
460 
461  else:
462  dist_indicator = True * np.ones((d, nr))
463 
464  elif method == 'dilate':
465  for i in range(nr):
466  A_temp = np.reshape(A[:, i].toarray(), dims[::-1])
467  if len(expandCore) > 0:
468  if len(expandCore.shape) < len(dims): # default for 3D
469  expandCore = iterate_structure(
470  generate_binary_structure(len(dims), 1), 2).astype(int)
471  A_temp = grey_dilation(A_temp, footprint=expandCore)
472  else:
473  A_temp = grey_dilation(A_temp, [1] * len(dims))
474 
475  dist_indicator[:, i] = np.squeeze(np.reshape(A_temp, (d, 1))) > 0
476  else:
477  dist_indicator = True * np.ones((d, nr))
478 
479  return dist_indicator

483 def construct_ellipse_parallel(pars):
484  """update spatial footprints and background through Basis Pursuit Denoising
485 
486  for each pixel i solve the problem
487  [A(i,:),b(i)] = argmin sum(A(i,:))
488  subject to
489  || Y(i,:) - A(i,:)*C + b(i)*f || <= sn(i)*sqrt(T);
490 
491  for each pixel the search is limited to a few spatial components
492 
493  Parameters:
494  ----------
495  [parsed]
496  cm[i]:
497  center of mass of each neuron
498 
499  A[:, i]: the A of each components
500 
501  Vr:
502 
503  dims:
504  the dimension of each A's ( same usually )
505 
506  dist:
507  computed distance matrix
508 
509  min_size: [optional] int
510 
511  max_size: [optional] int
512 
513  Returns:
514  --------
515  dist: np.ndarray
516  new estimate of spatial footprints
517 
518  Raise:
519  -------
520  Exception('You cannot pass empty (all zeros) components!')
521  """
522  Coor, cm, A_i, Vr, dims, dist, max_size, min_size, d = pars
523  dist_cm = coo_matrix(np.hstack([Coor[c].reshape(-1, 1) - cm[k]
524  for k, c in enumerate(['x', 'y', 'z'][:len(dims)])]))
525  Vr.append(dist_cm.T * spdiags(A_i.toarray().squeeze(),
526  0, d, d) * dist_cm / A_i.sum(axis=0))
527 
528  if np.sum(np.isnan(Vr)) > 0:
529  raise Exception('You cannot pass empty (all zeros) components!')
530 
531  D, V = eig(Vr[-1])
532 
533  dkk = [np.min((max_size ** 2, np.max((min_size ** 2, dd.real)))) for dd in D]
534 
535  # search indexes for each component
536  return np.sqrt(np.sum([old_div((dist_cm * V[:, k]) ** 2, dkk[k]) for k in range(len(dkk))], 0)) <= dist

712 def nnls_L0(X, Yp, noise):
713  """
714  Nonnegative least square with L0 penalty
715 
716  It will basically call the scipy function with some tests
717  we want to minimize :
718  min|| Yp-W_lam*X||**2 <= noise
719  with ||W_lam||_0 penalty
720  and W_lam >0

726 
727  Y: np.array
728  ((the regressand

730  Returns:
731  --------
732  W_lam: np.array
733  the learned weight matrices ((Models
734 
735  """
736  W_lam, RSS = scipy.optimize.nnls(X, np.ravel(Yp))
737  RSS = RSS * RSS
738  if RSS > noise: # hard noise constraint problem infeasible
739  return W_lam
740 
741  while 1:
742  eliminate = []

757 def lars_regression_noise_old(Yp, X, positive, noise, verbose=False):
758  """
759  Run LARS for regression problems with LASSO penalty, with optional positivity constraints
760  Author: Andrea Giovannucci. Adapted code from Eftychios Pnevmatikakis
761 
762 
763  Parameters:
764  -------
765  Yp: Yp[:,t] is the observed data at time t
766 
767  X: the regresion problem is Yp=X*W + noise
768 
769  maxcomps: maximum number of active components to allow
770 
771  positive: a flag to enforce positivity
772 
773  noise: the noise of the observation equation. if it is not
774  provided as an argument, the noise is computed from the
775  variance at the end point of the algorithm. The noise is
776  used in the computation of the Cp criterion.
777 
778  Returns:
779  -------
780  Ws: weights from each iteration
781  lambdas: lambda_ values at each iteration
782 
783  Cps: C_p estimates
784  last_break: last_break(m) == n means that the last break with m non-zero weights is at Ws(:,:,n)
785 
786  See Also:
787  -------
788  LARS : https://en.wikipedia.org/wiki/Least-angle_regression
789  group Lasso :
790  """
791  #INITAILIZATION
792  k = 1
793  Yp = np.squeeze(np.asarray(Yp))
794  # necessary for matrix multiplications
795  Yp = np.expand_dims(Yp, axis=1)
796  _, T = np.shape(Yp) # of time steps
797  _, N = np.shape(X) # of compartments
798 
799  # 1 : Start with all Weights equal to zero.
800  maxcomps = N
801  W = np.zeros((N, k))
802  active_set = np.zeros((N, k))
803  visited_set = np.zeros((N, k))
804  lambdas = []
805  # Just preallocation. Ws may end with more or less than maxcomp columns
806  Ws = []
807  r = np.expand_dims(np.dot(X.T, Yp.flatten()), axis=1) # N-dim vector
808  M = np.dot(-X.T, X) # N x N matrix
809 
810  # %% begin main loop
811  i = 0
812  flag = 0
813  while 1:
814  if flag == 1:
815  break
816  # % calculate new gradient component if necessary
817  if i > 0 and new >= 0 and visited_set[new] == 0: # AG NOT CLEAR HERE
818  visited_set[new] = 1 # % remember this direction was computed
819  # % Compute full gradient of Q
820  dQ = r + np.dot(M, W)
821 
822  # % Compute new W
823  if i == 0:
824  if positive:
825  dQa = dQ
826  else:
827  dQa = np.abs(dQ)
828  lambda_, new = np.max(dQa), np.argmax(dQa)
829 
830  if lambda_ < 0:
831  print('All negative directions!')
832  break
833  else:# 2 : Find the predictor x_{j} most correlated with y
834 
835  # % calculate vector to travel along
836  avec, gamma_plus, gamma_minus = calcAvec(new, dQ, W, lambda_, active_set, M,positive)
837  # % calculate time of travel and next new direction
838  if new == -1: # % if we just dropped a direction we don't allow it to emerge
839  if dropped_sign == 1: # % with the same sign
840  gamma_plus[dropped] = np.inf
841  else:
842  gamma_minus[dropped] = np.inf
843  # 3 : Increase the coefficient W in the direction of the sign of its correlation with y
844  gamma_plus[active_set == 1] = np.inf # % don't consider active components
845  gamma_plus[gamma_plus <= 0] = np.inf # % or components outside the range [0, lambda_]
846  gamma_plus[gamma_plus > lambda_] = np.inf
847  gp_min, gp_min_ind = np.min(gamma_plus), np.argmin(gamma_plus)
848 
849  if not positive:
850  gamma_minus[active_set == 1] = np.inf
851  gamma_minus[gamma_minus > lambda_] = np.inf
852  gamma_minus[gamma_minus <= 0] = np.inf
853  gm_min, gm_min_ind = np.min(gamma_minus), np.argmin(gamma_minus)
854 
855  [g_min, which] = np.min(gp_min), np.argmin(gp_min)
856 # % if there are no possible new components, try move to the end
857  if g_min == np.inf:
858  g_min = lambda_
859 # % This happens when all the components are already active or, if positive==1,
860 # when there are no new positive directions
861 
862  # % LARS check (is g_min*avec too large?)
863  gamma_zero = old_div(-W[active_set == 1], np.squeeze(avec))
864  gamma_zero_full = np.zeros((N, k))
865  gamma_zero_full[active_set == 1] = gamma_zero
866  gamma_zero_full[gamma_zero_full <= 0] = np.inf
867  gz_min, gz_min_ind = np.min(gamma_zero_full), np.argmin(gamma_zero_full)
868  #4: Increase Wj,Wk in their joint least squares direction, until some other predictor x_{m}
869  # has as much correlation with the residual r. (see 5)
870  if gz_min < g_min:
871  if verbose:
872  print(('DROPPING active weight:' + str(gz_min_ind)))
873  active_set[gz_min_ind] = 0
874  dropped = gz_min_ind
875  dropped_sign = np.sign(W[dropped])
876  W[gz_min_ind] = 0
877  avec = avec[gamma_zero != gz_min]
878  g_min = gz_min
879  new = -1
880 
881  elif g_min < lambda_:
882  if which == 0:
883  new = gp_min_ind
884  if verbose:
885  print(('new positive component:' + str(new)))
886 
887  else:
888  new = gm_min_ind
889  print(('new negative component:' + str(new)))
890 
891  W[active_set == 1] = W[active_set == 1] + np.dot(g_min, np.squeeze(avec))
892  if positive:
893  if any(W < 0):
894  flag = 1
895 
896  lambda_ = lambda_ - g_min
897 
898  # % Update weights and lambdas
899  lambdas.append(lambda_)
900  Ws.append(W.copy())
901  # 5 : Take residuals r=y-y_ along the way. Stop when some other predictor x_{k}
902  #has as much correlation with r as x_{j} has.
903  if len((Yp - np.dot(X, W)).shape) > 2:
904  res = scipy.linalg.norm(np.squeeze(Yp - np.dot(X, W)), 'fro') ** 2
905  else:
906  res = scipy.linalg.norm(Yp - np.dot(X, W), 'fro') ** 2
907 
908  # % Check finishing conditions
909  if lambda_ == 0 or (new >= 0 and np.sum(active_set) == maxcomps) or (res < noise):
910  if verbose:
911  print('end. \n')
912  break
913  # 6: Continue until: all predictors are in the model
914  if new >= 0:
915  active_set[new] = 1
916 
917  i = i + 1
918 
919  # end main loop
920 
921  # %% final calculation of mus
922  temp = Ws
923  Ws = np.asarray(np.swapaxes(np.swapaxes(temp, 0, 1), 1, 2))
924  if flag == 0:
925  if i > 0:
926  Ws = np.squeeze(Ws[:, :, :len(lambdas)])
927  w_dir = old_div(-(Ws[:, i] - Ws[:, i - 1]), (lambdas[i] - lambdas[i - 1]))
928  Aw = np.dot(X, w_dir)
929  y_res = np.squeeze(Yp) - np.dot(X, Ws[:, i - 1] + w_dir * lambdas[i - 1])
930  ld = scipy.roots([scipy.linalg.norm(Aw) ** 2, -2 * np.dot(Aw.T, y_res),
931  np.dot(y_res.T, y_res) - noise])
932  lam = ld[np.intersect1d(np.where(ld > lambdas[i]), np.where(ld < lambdas[i - 1]))]
933  if len(lam) == 0 or np.any(lam) < 0 or np.any(~np.isreal(lam)):
934  lam = np.array([lambdas[i]])
935 
936  W_lam = Ws[:, i - 1] + np.dot(w_dir, lambdas[i - 1] - lam[0])
937  else:
938  warn('LARS REGRESSION NOT SOLVABLE, USING NN LEAST SQUARE')
939  W_lam = scipy.optimize.nnls(X, np.ravel(Yp))[0]
940  lam = 10
941 
942  else:
943  W_lam = 0
944  Ws = 0
945  lambdas = 0
946  lam = 0
947 
948  return Ws, lambdas, W_lam, lam, flag

952 def calcAvec(new, dQ, W, lambda_, active_set, M,positive):
953  """
954  calculate the vectoe to travel along
955 
956  used in the lars regression function
957 
958  Parameters:
959  ----------
960  Y: np.ndarray (2D or 3D)
961  movie, raw data in 2D or 3D (pixels x time).
962 
963  Cf: np.ndarray

965 
966  Returns:
967  --------
968  C_name: string
969  the memmaped name of Cf
970 
971  Y_name: string
972  the memmaped name of Y
973 
974 
975  """
976  r, c = np.nonzero(active_set)
977  Mm = -M.take(r, axis=0).take(r, axis=1)
978  Mm = old_div((Mm + Mm.T), 2)
979  # % verify that there is no numerical instability
980  if len(Mm) > 1:
981  eigMm, _ = scipy.linalg.eig(Mm)
982  eigMm = np.real(eigMm)
983  else:
984  eigMm = Mm
985  if any(eigMm < 0):
986  np.min(eigMm)
987 
988  b = np.sign(W)
989  if new >= 0:
990  b[new] = np.sign(dQ[new])
991  b = b[active_set == 1]
992  if len(Mm) > 1:

1002  gamma_minus = old_div((lambda_ + dQ), (one_vec - dQa))
1003 
1004  return avec, gamma_plus, gamma_minus
