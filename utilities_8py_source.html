326 def extract_DF_F(Yr, A, C, bl, quantileMin = 8, frames_window = 200, block_size = 400, dview = None):
327  """ Compute DFF function from cnmf output.
328 
329  Disclaimer: it might be memory inefficient
330 
331  Parameters:
332  -----------
333  Yr: ndarray (2D)
334  movie pixels X time
335 
336  A: scipy.sparse.coo_matrix
337  spatial components (from cnmf cnm.A)
338 
339  C: ndarray
340  temporal components (from cnmf cnm.C)
341 
342  bl: ndarray
343  baseline for each component (from cnmf cnm.bl)
344 
345  quantile_min: float
346  quantile minimum of the
347 
348  frammes_window: int
349  number of frames for running quantile
350 
351  Returns:
352  -------
353 
354  Cdf:
355  the computed Calcium acitivty to the derivative of f
356 
357  See Also:
358  -------
359 
360  ..image::docs/img/onlycnmf.png
361 
362  """
363  nA = np.array(np.sqrt(A.power(2).sum(0)).T)
364  A = scipy.sparse.coo_matrix(A/nA.T)
365  C = C*nA
366  bl = (bl*nA.T).squeeze()
367  nA = np.array(np.sqrt(A.power(2).sum(0)).T)
368 
369 
370  T = C.shape[-1]
371  if 'memmap' in str(type(Yr)):
372  if block_size >= 500:
373  print('Forcing single thread for memory issues')
374  dview_res = None
375  else:
376  print('Using thread. If memory issues set block_size larger than 500')
377  dview_res = dview
378 
379  AY = parallel_dot_product(Yr, A, dview=dview_res, block_size=block_size,
380  transpose=True).T
381  else:
382  AY = A.T.dot(Yr)
383 
384  bas_val = bl[None,:]
385  Bas = np.repeat(bas_val,T,0).T
386  AA = A.T.dot(A)
387  AA.setdiag(0)
388  Cf = (C-Bas)*(nA**2)
389  C2 = AY - AA.dot(C)
390 
391 
392  if frames_window is None or frames_window > T:
393  Df = np.percentile(C2,quantileMin, axis = 1)
394  C_df = Cf/Df[:,None]
395 
396  else:
397  Df = scipy.ndimage.percentile_filter(C2, quantileMin, (frames_window,1))
398  C_df = Cf/Df
399 
400  return C_df

535 def app_vertex_cover(A):
536  """ Finds an approximate vertex cover for a symmetric graph with adjacency matrix A.
537 
538  Parameters:
539  -----------
540  A: boolean 2d array (K x K)
541  Adjacency matrix. A is boolean with diagonal set to 0
542 
543  Returns:
544  --------
545  L: A vertex cover of A
546 
547  @authors by Eftychios A. Pnevmatikakis, Simons Foundation, 2015
548  """
549 
550  L = []
551  while A.any():
552  nz = np.nonzero(A)[0] # find non-zero edges
553  u = nz[np.random.randint(0, len(nz))]
554  A[u, :] = False
555  A[:, u] = False
556  L.append(u)
557 
558  return np.asarray(L)

561 def update_order(A):
562  """Determines the update order of the temporal components
563 
564  this, given the spatial components, by creating a nest of random approximate vertex covers
565  Basically we can update the components that are not overlapping, in parallel
566 
567  Input:
568  -------
569  A: np.ndarray
570  matrix of spatial components (d x K)
571 
572  Outputs:
573  ---------
574  parllcomp: list of sets
575  list of subsets of components. The components of each subset can be updated in parallel
576 
577  len_parrllcomp: list
578  length of each subset
579 
580  @authors: Eftychios A. Pnevmatikakis, Simons Foundation, 2015
581  """
582  K = np.shape(A)[-1]
583  AA = A.T * A
584  AA.setdiag(0)
585  F = (AA) > 0
586  F = F.toarray()
587  rem_ind = np.arange(K)
588  parllcomp = []
589  len_parrllcomp = []
590  while len(rem_ind) > 0:
591  L = np.sort(app_vertex_cover(F[rem_ind, :][:, rem_ind]))
592  if L.size:
593  ord_ind = set(rem_ind) - set(rem_ind[L])
594  rem_ind = rem_ind[L]
595  else:
596  ord_ind = set(rem_ind)
597  rem_ind = []
598 
599  parllcomp.append(ord_ind)
600  len_parrllcomp.append(len(ord_ind))
601 
602  return parllcomp[::-1], len_parrllcomp[::-1]

605 def update_order_greedy(A,flag_AA = True):
606  """Determines the update order of the temporal components
607 
608  this, given the spatial components using a greedy method
609  Basically we can update the components that are not overlapping, in parallel
610 
611  Input:
612  -------
613  A: sparse crc matrix
614  matrix of spatial components (d x K)
615  OR
616  A.T.dot(A) matrix (d x d) if flag_AA = true
617 
618  flag_AA: boolean (default true)
619 
620  Outputs:
621  ---------
622  parllcomp: list of sets
623  list of subsets of components. The components of each subset can be updated in parallel
624 
625  len_parrllcomp: list
626  length of each subset
627 
628  @author: Eftychios A. Pnevmatikakis, Simons Foundation, 2017
629  """
630  K = np.shape(A)[-1]
631  parllcomp = []
632  for i in range(K):
633  new_list = True
634  for ls in parllcomp:
635  if flag_AA:
636  if A[i,ls].nnz == 0:
637  ls.append(i)
638  new_list = False
639  break
640  else:
641  if (A[:,i].T.dot(A[:,ls])).nnz == 0:
642  ls.append(i)
643  new_list = False
644  break
645 
646  if new_list:
647  parllcomp.append([i])
648  len_parrllcomp = [len(ls) for ls in parllcomp]
649  return parllcomp,len_parrllcomp
