231  def extract_shifts(self, max_shift_w=5,max_shift_h=5, template=None, method='opencv'):
232  """
233  Performs motion corretion using the opencv matchtemplate function. At every iteration a template is built by taking the median of all frames and then used to align the other frames.
234 
235  Parameters:
236  ----------
237  max_shift_w,max_shift_h: maximum pixel shifts allowed when correcting in the width and height direction
238 
239  template: if a good template for frame by frame correlation is available it can be passed. If None it is automatically computed
240

242

244  -------

258  self=self-min_val

263  warnings.warn('Casting the array to float 32')

265

273  else:

275  warnings.warn('Pixels averages are too negative for template. Removing 1 percentile.')
276  template=template-np.percentile(template,1)

278  template=template[ms_h:h_i-ms_h,ms_w:w_i-ms_w].astype(np.float32)
279 
280  #% run algorithm, press q to stop it
281  shifts=[] # store the amount of shift in each frame
282  xcorrs=[]
283 
284  for i,frame in enumerate(self):
285  if i%100==99:
286  print(("Frame %i"%(i+1)))
287  if method == 'opencv':
288  res = cv2.matchTemplate(frame,template,cv2.TM_CCORR_NORMED)
289  top_left = cv2.minMaxLoc(res)[3]
290  elif method == 'skimage':
291  res = match_template(frame,template)
292  top_left = np.unravel_index(np.argmax(res),res.shape)
293  top_left=top_left[::-1]

296  avg_corr=np.mean(res)
297  sh_y,sh_x = top_left

300  # if max is internal, check for subpixel shift using gaussian
301  # peak registration

303  log_xp1_y = np.log(res[sh_x+1,sh_y])
304  log_x_ym1 = np.log(res[sh_x,sh_y-1])

307 
308  sh_x_n = -(sh_x - ms_h + old_div((log_xm1_y - log_xp1_y), (2 * log_xm1_y - four_log_xy + 2 * log_xp1_y)))
309  sh_y_n = -(sh_y - ms_w + old_div((log_x_ym1 - log_x_yp1), (2 * log_x_ym1 - four_log_xy + 2 * log_x_yp1)))
310  else:
311  sh_x_n = -(sh_x - ms_h)
312  sh_y_n = -(sh_y - ms_w)
313 
314  shifts.append([sh_x_n,sh_y_n])
315  xcorrs.append([avg_corr])
316 
317  self=self+min_val
318

582  def IPCA(self, components = 50, batch =1000):
583  """
584  Iterative Principal Component analysis, see sklearn.decomposition.incremental_pca
585  Parameters:
586  ------------
587  components (default 50) = number of independent components to return
588 
589  batch (default 1000) = number of pixels to load into memory simultaneously in IPCA. More requires more memory but leads to better fit
590 
591  Returns:
592  -------
593  eigenseries: principal components (pixel time series) and associated singular values
594 
595  eigenframes: eigenframes are obtained by multiplying the projected frame matrix by the projected movie (whitened frames?)
596 
597  proj_frame_vectors:the reduced version of the movie vectors using only the principal component projection
598  """
599  # vectorize the images
600  num_frames, h, w = np.shape(self)
601  frame_size = h * w
602  frame_samples = np.reshape(self, (num_frames, frame_size)).T
603 
604  # run IPCA to approxiate the SVD
605  ipca_f = incremental_pca(n_components=components, batch_size=batch)
606  ipca_f.fit(frame_samples)
607 
608  # construct the reduced version of the movie vectors using only the
609  # principal component projection
610 
611  proj_frame_vectors = ipca_f.inverse_transform(ipca_f.transform(frame_samples))
612 
613  # get the temporal principal components (pixel time series) and
614  # associated singular values
615 
616  eigenseries = ipca_f.components_.T
617 
618  # the rows of eigenseries are approximately orthogonal
619  # so we can approximately obtain eigenframes by multiplying the
620  # projected frame matrix by this transpose on the right
621 
622  eigenframes = np.dot(proj_frame_vectors, eigenseries)
623 
624  return eigenseries, eigenframes, proj_frame_vectors

632  Parameters:
633  -----------
634  components (default 50) = number of independent components to return
635 
636  batch (default 1000) = number of pixels to load into memory simultaneously in IPCA. More requires more memory but leads to better fit
