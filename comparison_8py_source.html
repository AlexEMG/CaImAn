443 def cnmf(Cn,A_gt, A_test,C_gt,C_test, dims_gt, dims_test, dview= None, sensitivity =0,timer=0,n_frames_per_bin = 10):
444 
445  A_test=A_test.toarray() #coo sparse matrix
446  A_gt = A_gt.toarray()
447 
448  #proceed to a trhreshold
449  A_test_thr = cm.source_extraction.cnmf.spatial.threshold_components(
450  A_test, dims_test, medw=None, thr_method='max', maxthr=0.2, nrgthr=0.99, extract_cc=True,
451  se=None, ss=None, dview=dview)
452  A_gt_thr = cm.source_extraction.cnmf.spatial.threshold_components(
453  A_gt, dims_gt, medw=None, thr_method='max', maxthr=0.2, nrgthr=0.99, extract_cc=True,
454  se=None, ss=None, dview=dview)
455 
456  #compute C using this A thr
457  A_test_thr = A_test_thr > 0
458  A_gt_thr = A_gt_thr > 0
459  #we do not compute a threshold on the size of neurons
460  C_test_thr = C_test
461  C_gt_thr = C_gt
462  #we would also like the difference in the number of neurons
463  diffneur = A_test_thr.shape[1] - A_gt_thr.shape[1]
464  print(diffneur)
465  #computing the values
466  C_test_thr = np.array([CC.reshape([-1,n_frames_per_bin]).max(1) for CC in C_test_thr])
467  C_gt_thr = np.array([CC.reshape([-1,n_frames_per_bin]).max(1) for CC in C_gt_thr])
468  maskgt = A_gt_thr[:,:].reshape([dims_gt[0],dims_gt[1],-1],order = 'F').transpose([2,0,1])*1.
469  masktest = A_test_thr[:,:].reshape([dims_test[0],dims_test[1],-1],order = 'F').transpose([2,0,1])*1.
470 
471  idx_tp_gt,idx_tp_comp, idx_fn_gt, idx_fp_comp, performance_off_on = \
472  cm.base.rois.nf_match_neurons_in_binary_masks(masks_gt=maskgt,
473  masks_comp = masktest, Cn=Cn, plot_results=True)
474 
475  #the pearson's correlation coefficient of the two Calcium activities thresholded
476  #comparing Calcium activities of all the components that are defined by the matching algo as the same.
477  corrs = np.array([scipy.stats.pearsonr(
478  C_gt_thr[gt,:],C_test_thr[comp,:])[0] for gt,comp in zip(idx_tp_gt,idx_tp_comp)])
479  #todo, change this test when I will have found why I have one additionnal neuron
480  isdiff = True if (diffneur != 0 and (np.linalg.norm(corrs) < sensitivity))else False
481  info= {'isdifferent':int(isdiff),
482  'diff_data': { 'performance':performance_off_on,
483  'corelations':corrs.tolist(),
484  #performance = dict()
485  #performance['recall'] = old_div(TP,(TP+FN))
486  #performance['precision'] = old_div(TP,(TP+FP))
487  #performance['accuracy'] = old_div((TP+TN),(TP+FP+FN+TN))
488  #performance['f1_score'] = 2*TP/(2*TP+FP+FN)
489  'diffneur':diffneur},
490  'diff_timing': timer}
491  return info
