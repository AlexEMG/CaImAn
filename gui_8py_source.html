10 def pretty_roi(mask):
11  mask = mask.astype(np.float)
12  colors = np.arange(1,len(mask)+1)
13  mask *= colors[...,np.newaxis,np.newaxis]
14  mask = np.max(mask, axis=0)
15  mask[mask==0] = None
16  return mask

18 def isin(r, ax):
19  x0,x1,y0,y1 = ax.axis()
20  xmin,xmax = min(x0,x1),max(x0,x1)
21  ymin,ymax = min(y0,y1),max(y0,y1)
22  return (ymin<r[0]<ymax) and (xmin<r[1]<xmax)
